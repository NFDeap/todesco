
##	Comandos Auxiliares		##

	$ php artisan migrate	//Preenche o arquivo database.sqlite com as tabelas.
	$ php artisan make:seed /*EX: UsuarioSeeds*/ // Cria o Seeds.
	$ php artisan db:seed // Popula o banco.

	$ php artisan serve	//Roda o servidor local.
